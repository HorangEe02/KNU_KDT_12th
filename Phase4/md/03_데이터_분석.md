## 3. 상관분석 (Correlation Analysis)

### 3.1 이론적 배경

**연구 가설:**
- H₀: 기온 변화와 소스 원료 생산량 사이에 상관관계가 없다 (ρ = 0)
- H₁: 기온 변화와 소스 원료 생산량 사이에 상관관계가 있다 (ρ ≠ 0)

**분석 방법:**
- Pearson 상관계수: 선형 관계의 강도와 방향
- Spearman 상관계수: 단조(monotonic) 관계 (비모수)

### 3.2 기온-생산량 상관분석

```python
def sauce_correlation_analysis(data):
    """
    소스 원료별 기온-생산량 상관분석
    """
    results = []
    
    print("\n" + "=" * 80)
    print("🍅🥚🌶️ 기온 변화 vs 소스 원료 생산량 상관분석")
    print("=" * 80)
    print(f"{'원료':<25} {'Pearson r':>12} {'p-value':>12} {'Spearman ρ':>12} {'p-value':>12} {'N':>6}")
    print("-" * 80)
    
    items_info = [
        ('Tomatoes', '토마토 (케찹)'),
        ('Eggs', '계란 (마요네즈)'),
        ('Chillies_dry', '건고추 (스리라차)')
    ]
    
    for key, name in items_info:
        if key in data and 'World_Total' in data[key].columns and 'Temp_Anomaly' in data[key].columns:
            df = data[key].dropna(subset=['World_Total', 'Temp_Anomaly'])
            
            x = df['Temp_Anomaly'].values
            y = df['World_Total'].values
            n = len(x)
            
            # Pearson
            pearson_r, pearson_p = pearsonr(x, y)
            
            # Spearman
            spearman_r, spearman_p = spearmanr(x, y)
            
            sig_p = "***" if pearson_p < 0.001 else "**" if pearson_p < 0.01 else "*" if pearson_p < 0.05 else ""
            sig_s = "***" if spearman_p < 0.001 else "**" if spearman_p < 0.01 else "*" if spearman_p < 0.05 else ""
            
            print(f"{name:<25} {pearson_r:>10.4f}{sig_p:<2} {pearson_p:>10.4f} {spearman_r:>10.4f}{sig_s:<2} {spearman_p:>10.4f} {n:>6}")
            
            results.append({
                'Item': name,
                'Item_Key': key,
                'N': n,
                'Pearson_r': pearson_r,
                'Pearson_p': pearson_p,
                'Pearson_sig': pearson_p < ALPHA,
                'Spearman_r': spearman_r,
                'Spearman_p': spearman_p,
                'Spearman_sig': spearman_p < ALPHA
            })
    
    print("-" * 80)
    print("유의수준: *** p<0.001, ** p<0.01, * p<0.05")
    
    return pd.DataFrame(results)


df_corr = sauce_correlation_analysis(data)
df_corr.to_csv(OUTPUT_DIR / 'sauce_stat_01_correlation.csv', index=False)
```

## 4. 기후 민감도 분석 (Climate Sensitivity)

### 4.1 변동성 비교

```python
def climate_sensitivity_analysis(data):
    """
    소스 원료별 기후 민감도 비교
    
    기후 민감도 지표:
    1. 생산량 변동성 (Coefficient of Variation)
    2. 연간 변화율의 표준편차
    3. 기온 변화에 대한 민감도 (회귀 기울기)
    """
    
    print("\n" + "=" * 70)
    print("🌡️ 소스 원료별 기후 민감도 분석")
    print("=" * 70)
    
    results = []
    
    items = [
        ('Tomatoes', '토마토 (케찹)'),
        ('Eggs', '계란 (마요네즈)'),
        ('Chillies_dry', '건고추 (스리라차)')
    ]
    
    for key, name in items:
        if key not in data or 'World_Total' not in data[key].columns:
            continue
        
        df = data[key]
        production = df['World_Total']
        
        # 1. 변동계수 (CV)
        cv = (production.std() / production.mean()) * 100
        
        # 2. 연간 변화율 표준편차
        yoy_change = production.pct_change() * 100
        yoy_std = yoy_change.std()
        
        # 3. 기온 민감도 (회귀 기울기)
        if 'Temp_Anomaly' in df.columns:
            df_clean = df.dropna(subset=['World_Total', 'Temp_Anomaly'])
            X = sm.add_constant(df_clean['Temp_Anomaly'])
            y = df_clean['World_Total'] / 1e6  # 백만 톤
            model = sm.OLS(y, X).fit()
            temp_sensitivity = model.params['Temp_Anomaly']
            temp_sensitivity_pct = (temp_sensitivity / y.mean()) * 100  # 평균 대비 %
        else:
            temp_sensitivity = np.nan
            temp_sensitivity_pct = np.nan
        
        print(f"\n[{name}]")
        print(f"   변동계수 (CV): {cv:.2f}%")
        print(f"   연간 변화율 SD: {yoy_std:.2f}%")
        print(f"   기온 민감도: {temp_sensitivity:.2f} MT/°C ({temp_sensitivity_pct:.2f}%/°C)")
        
        results.append({
            'Item': name,
            'CV_Percent': cv,
            'YoY_SD': yoy_std,
            'Temp_Sensitivity_MT_per_C': temp_sensitivity,
            'Temp_Sensitivity_Pct_per_C': temp_sensitivity_pct
        })
    
    df_results = pd.DataFrame(results)
    
    # 민감도 순위
    print("\n" + "-" * 50)
    print("📊 기후 민감도 순위 (변동계수 기준)")
    df_ranked = df_results.sort_values('CV_Percent', ascending=False)
    for i, (_, row) in enumerate(df_ranked.iterrows(), 1):
        print(f"   {i}. {row['Item']}: CV = {row['CV_Percent']:.2f}%")
    
    return df_results


df_sensitivity = climate_sensitivity_analysis(data)
df_sensitivity.to_csv(OUTPUT_DIR / 'sauce_stat_02_sensitivity.csv', index=False)
```

## 5. 회귀분석 (Regression Analysis)

### 5.1 단순 선형 회귀분석

```python
def sauce_regression_analysis(data):
    """
    소스 원료별 회귀분석
    
    모델: Production = β₀ + β₁ × Temperature_Anomaly + ε
    """
    
    print("\n" + "=" * 80)
    print("📈 회귀분석: 기온 변화 → 소스 원료 생산량")
    print("=" * 80)
    
    results = []
    
    items = [
        ('Tomatoes', '토마토 (케찹)'),
        ('Eggs', '계란 (마요네즈)'),
        ('Chillies_dry', '건고추 (스리라차)')
    ]
    
    fig, axes = plt.subplots(1, 3, figsize=(16, 5))
    colors = ['#E74C3C', '#F1C40F', '#C0392B']
    
    for idx, ((key, name), ax, color) in enumerate(zip(items, axes, colors)):
        if key not in data:
            continue
        
        df = data[key].dropna(subset=['World_Total', 'Temp_Anomaly'])
        
        X = df['Temp_Anomaly'].values
        y = df['World_Total'].values / 1e6  # 백만 톤
        
        # OLS 회귀분석
        X_const = sm.add_constant(X)
        model = sm.OLS(y, X_const).fit()
        
        beta0 = model.params[0]  # 절편
        beta1 = model.params[1]  # 기울기
        r_squared = model.rsquared
        f_pval = model.f_pvalue
        
        # 잔차 분석
        residuals = model.resid
        
        # 정규성 검정
        if len(residuals) <= 50:
            _, shapiro_p = shapiro(residuals)
        else:
            shapiro_p = np.nan
        
        # 등분산성 검정
        _, bp_pval, _, _ = het_breuschpagan(residuals, X_const)
        
        print(f"\n[{name}]")
        print(f"   회귀식: Production = {beta1:.3f} × Temp + {beta0:.3f}")
        print(f"   R² = {r_squared:.4f}")
        print(f"   F-test p-value = {f_pval:.4e}")
        print(f"   해석: 기온 1°C 상승 시 생산량 {beta1:.3f} 백만톤 {'증가' if beta1 > 0 else '감소'}")
        print(f"   [가정 검정]")
        print(f"   - 잔차 정규성: p = {shapiro_p:.4f} {'✓' if shapiro_p > 0.05 else '✗'}")
        print(f"   - 등분산성: p = {bp_pval:.4f} {'✓' if bp_pval > 0.05 else '✗'}")
        
        # 시각화
        ax.scatter(X, y, c=color, s=60, alpha=0.7, edgecolors='black', linewidth=0.5)
        
        # 회귀선
        x_line = np.linspace(X.min(), X.max(), 100)
        y_line = beta0 + beta1 * x_line
        ax.plot(x_line, y_line, 'k-', linewidth=2)
        
        # 95% 신뢰구간
        y_pred = model.predict(X_const)
        se = np.sqrt(np.sum(residuals**2) / (len(X) - 2))
        ax.fill_between(x_line, y_line - 1.96*se, y_line + 1.96*se, alpha=0.2, color='gray')
        
        # 통계량 표시
        sig = "***" if f_pval < 0.001 else "**" if f_pval < 0.01 else "*" if f_pval < 0.05 else "ns"
        ax.text(0.05, 0.95, f'y = {beta1:.2f}x + {beta0:.2f}\nR² = {r_squared:.3f} ({sig})',
               transform=ax.transAxes, fontsize=10, verticalalignment='top',
               bbox=dict(boxstyle='round', facecolor='white', alpha=0.8))
        
        ax.set_xlabel('Temperature Anomaly (°C)')
        ax.set_ylabel('Production (Million Tonnes)')
        ax.set_title(name, fontweight='bold')
        ax.grid(alpha=0.3)
        
        results.append({
            'Item': name,
            'Intercept': beta0,
            'Slope': beta1,
            'R_squared': r_squared,
            'F_pvalue': f_pval,
            'Shapiro_p': shapiro_p,
            'BP_p': bp_pval
        })
    
    plt.suptitle('Linear Regression: Temperature → Sauce Ingredient Production\n'
                 'Source: FAOSTAT', fontsize=13, fontweight='bold', y=1.02)
    plt.tight_layout()
    plt.savefig(OUTPUT_DIR / 'sauce_stat_04_regression.png', dpi=150, bbox_inches='tight')
    plt.close()
    
    print("\n✅ 회귀분석 시각화 저장")
    
    return pd.DataFrame(results)


df_reg = sauce_regression_analysis(data)
df_reg.to_csv(OUTPUT_DIR / 'sauce_stat_04_regression.csv', index=False)
```

```python
def sriracha_crisis_analysis(data):
    """
    2022년 스리라차 품귀 사태 분석
    
    배경:
    - 2022년 미국 Huy Fong Foods社 스리라차 생산 중단
    - 원인: 멕시코산 할라피뇨 고추 공급 차질
    - 멕시코 뉴멕시코/캘리포니아 지역 극심한 가뭄
    """
    
    print("\n" + "=" * 70)
    print("🌶️ 2022년 스리라차 위기 분석 (Event Study)")
    print("=" * 70)
    
    if 'Chillies_dry' not in data:
        print("⚠️ 고추 데이터 없음")
        return None
    
    df = data['Chillies_dry']
    
    # 이벤트 연도 정의
    event_year = 2022
    pre_years = [2019, 2020, 2021]
    post_years = [2022, 2023]
    
    print(f"\n[이벤트 정의]")
    print(f"   - 이벤트 연도: {event_year}")
    print(f"   - 사전 기간: {pre_years}")
    print(f"   - 사후 기간: {post_years}")
    
    # 기간별 생산량
    pre_prod = df[df['Year'].isin(pre_years)]['World_Total'].mean()
    event_prod = df[df['Year'] == event_year]['World_Total'].values[0]
    
    # 변화율
    change_pct = ((event_prod - pre_prod) / pre_prod) * 100
    
    print(f"\n[전체 건고추 생산량 분석]")
    print(f"   - 사전 기간 평균: {pre_prod/1e6:.2f} 백만톤")
    print(f"   - 2022년 생산량: {event_prod/1e6:.2f} 백만톤")
    print(f"   - 변화: {change_pct:+.2f}%")
    
    # 멕시코 분석 (스리라차 원료 공급국)
    if 'Mexico' in df.columns:
        pre_mexico = df[df['Year'].isin(pre_years)]['Mexico'].mean()
        event_mexico = df[df['Year'] == event_year]['Mexico'].values[0]
        mexico_change = ((event_mexico - pre_mexico) / pre_mexico) * 100
        
        print(f"\n[멕시코 건고추 생산량 분석]")
        print(f"   - 사전 기간 평균: {pre_mexico/1e6:.4f} 백만톤")
        print(f"   - 2022년 생산량: {event_mexico/1e6:.4f} 백만톤")
        print(f"   - 변화: {mexico_change:+.2f}%")
        print(f"   ⚠️ 멕시코 고추 생산 급감 → 스리라차 공급 위기")
    
    # 기온 분석
    if 'Temperature' in data and 'Mexico' in data['Temperature'].columns:
        df_temp = data['Temperature']
        pre_temp = df_temp[df_temp['Year'].isin(pre_years)]['Mexico'].mean()
        event_temp = df_temp[df_temp['Year'] == event_year]['Mexico'].values[0]
        
        print(f"\n[멕시코 기온 변화]")
        print(f"   - 사전 기간 평균 편차: {pre_temp:.3f}°C")
        print(f"   - 2022년 편차: {event_temp:.3f}°C")
        print(f"   - 차이: {event_temp - pre_temp:+.3f}°C (이상고온)")
    
    # 통계적 유의성 검정 (t-test)
    pre_data = df[df['Year'].isin(pre_years)]['World_Total'].values
    post_data = df[df['Year'].isin(post_years)]['World_Total'].values
    
    if len(pre_data) >= 2 and len(post_data) >= 2:
        t_stat, t_pval = ttest_ind(pre_data, post_data)
        print(f"\n[통계적 검정: 사전 vs 사후 생산량]")
        print(f"   - t-statistic: {t_stat:.3f}")
        print(f"   - p-value: {t_pval:.4f}")
        print(f"   - 결론: {'유의미한 차이' if t_pval < 0.05 else '유의미한 차이 없음'}")
    
    return {
        'Event': 'Sriracha Crisis 2022',
        'Pre_Mean': pre_prod,
        'Event_Value': event_prod,
        'Change_Pct': change_pct
    }


sriracha_result = sriracha_crisis_analysis(data)
```

```python
def extreme_event_analysis(data):
    """
    주요 이상기후 이벤트가 소스 원료에 미친 영향
    """
    
    # 주요 이상기후 이벤트
    events = {
        2012: {'name': 'US Drought', 'affected': ['Tomatoes', 'Eggs']},
        2015: {'name': 'Avian Influenza (AI)', 'affected': ['Eggs']},
        2019: {'name': 'European Heatwave', 'affected': ['Tomatoes']},
        2022: {'name': 'Sriracha Crisis (Mexico Drought)', 'affected': ['Chillies_dry']}
    }
    
    print("\n" + "=" * 70)
    print("🌪️ 이상기후 이벤트 영향 분석")
    print("=" * 70)
    
    results = []
    
    for year, event_info in events.items():
        print(f"\n[{year}: {event_info['name']}]")
        
        for item in event_info['affected']:
            if item not in data or 'World_Total' not in data[item].columns:
                continue
            
            df = data[item]
            
            # 해당 연도 및 전년도 데이터
            current = df[df['Year'] == year]['World_Total'].values
            previous = df[df['Year'] == year - 1]['World_Total'].values
            
            if len(current) == 0 or len(previous) == 0:
                continue
            
            change = ((current[0] - previous[0]) / previous[0]) * 100
            
            item_name = {
                'Tomatoes': '토마토 (케찹)',
                'Eggs': '계란 (마요네즈)',
                'Chillies_dry': '건고추 (스리라차)'
            }.get(item, item)
            
            print(f"   - {item_name}: {change:+.2f}%")
            
            results.append({
                'Year': year,
                'Event': event_info['name'],
                'Item': item_name,
                'Change_Pct': change
            })
    
    return pd.DataFrame(results)
```

```python
def korea_import_risk_analysis(data):
    """
    한국의 소스 원료 수입 리스크 분석
    
    분석 항목:
    1. 수입 의존도 변화
    2. 수입 변동성
    3. 기후 리스크 노출도
    """
    
    print("\n" + "=" * 70)
    print("🇰🇷 한국 소스 원료 수입 리스크 분석")
    print("=" * 70)
    
    if 'Korea_Import' not in data:
        print("⚠️ 한국 수입 데이터 없음")
        return None
    
    df = data['Korea_Import']
    
    items_info = [
        ('Tomato paste', '토마토 페이스트 (케찹)'),
        ('Eggs, hen, in shell', '계란 (마요네즈)'),
        ('Chillies and peppers, dry', '건고추 (스리라차)')
    ]
    
    results = []
    
    for col, name in items_info:
        if col not in df.columns:
            continue
        
        values = df[col]
        
        # 수입량 변화
        start = values.iloc[0]
        end = values.iloc[-1]
        total_change = ((end - start) / start) * 100
        
        # 변동성
        cv = (values.std() / values.mean()) * 100
        yoy_std = values.pct_change().std() * 100
        
        # CAGR (연평균 성장률)
        n_years = len(values) - 1
        cagr = ((end / start) ** (1/n_years) - 1) * 100
        
        print(f"\n[{name}]")
        print(f"   수입량 변화: {start/1e3:.1f}K → {end/1e3:.1f}K tonnes ({total_change:+.1f}%)")
        print(f"   연평균 성장률 (CAGR): {cagr:.2f}%")
        print(f"   변동계수 (CV): {cv:.2f}%")
        print(f"   연간 변화율 SD: {yoy_std:.2f}%")
        
        # 리스크 평가
        if cv > 20:
            risk_level = "🔴 고위험"
        elif cv > 10:
            risk_level = "🟡 중위험"
        else:
            risk_level = "🟢 저위험"
        
        print(f"   리스크 수준: {risk_level}")
        
        results.append({
            'Item': name,
            'Total_Change_Pct': total_change,
            'CAGR_Pct': cagr,
            'CV_Pct': cv,
            'YoY_SD_Pct': yoy_std,
            'Risk_Level': risk_level
        })
    
    return pd.DataFrame(results)


df_korea_risk = korea_import_risk_analysis(data)
if df_korea_risk is not None:
    df_korea_risk.to_csv(OUTPUT_DIR / 'sauce_stat_06_korea_risk.csv', index=False)
```

```python
def sauce_correlation_heatmap(data):
    """
    소스 원료 간 상관행렬 시각화
    """
    
    # 통합 데이터셋 생성
    df_integrated = data.get('Integrated')
    
    if df_integrated is None:
        print("⚠️ 통합 데이터셋 없음")
        return None
    
    # 피벗 (연도 × 품목)
    df_pivot = df_integrated.pivot(index='Year', columns='Item', values='World_Production_Tonnes')
    
    # 기온 데이터 추가
    if 'Temperature' in data and 'World' in data['Temperature'].columns:
        df_temp = data['Temperature'][['Year', 'World']].rename(columns={'World': 'Temperature'})
        df_pivot = df_pivot.reset_index().merge(df_temp, on='Year').set_index('Year')
    
    # 상관행렬
    corr_matrix = df_pivot.corr()
    
    # 히트맵
    fig, ax = plt.subplots(figsize=(10, 8))
    
    mask = np.triu(np.ones_like(corr_matrix, dtype=bool), k=1)
    
    sns.heatmap(corr_matrix,
                annot=True,
                fmt='.3f',
                cmap='RdBu_r',
                center=0,
                square=True,
                linewidths=0.5,
                mask=mask,
                vmin=-1, vmax=1,
                cbar_kws={'label': 'Correlation'},
                ax=ax)
    
    ax.set_title('Correlation Matrix: Sauce Ingredients & Temperature\n'
                 'Source: FAOSTAT', fontsize=13, fontweight='bold')
    
    plt.tight_layout()
    plt.savefig(OUTPUT_DIR / 'sauce_stat_07_heatmap.png', dpi=150, bbox_inches='tight')
    plt.close()
    
    print("\n✅ 상관행렬 히트맵 저장")
    print("\n📊 상관행렬:")
    print(corr_matrix.round(3))
    
    return corr_matrix
```
