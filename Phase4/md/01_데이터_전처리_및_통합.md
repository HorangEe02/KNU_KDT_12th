### 1.1 소스류 특성

**케찹 (토마토)**
- 전 세계적으로 가장 많이 소비되는 소스
- 토마토 가공품 중 대표 상품
- 기후 민감도: 중간 (관개 시설 의존)

**마요네즈 (계란)**

- 산란계 사육은 사료 가격(옥수수, 콩)에 민감
- 기후 민감도: 간접적 (사료 작물 영향)

**스리라차 (고추)**
- 2022년 스리라차 품귀 현상으로 주목
- 미국 Huy Fong Foods의 원료 공급 차질
- 기후 민감도: 매우 높음 (가뭄, 폭염에 취약)

```python
def create_tomato_production_sample():
    """
    토마토 생산량 샘플 데이터 생성
    
    출처: FAO. 2024. FAOSTAT: Crops and livestock products.
          https://www.fao.org/faostat/en/#data/QCL
    
    주요 생산국 (2022년 기준):
    - 중국: 약 68백만 톤 (세계 1위, 35%)
    - 인도: 약 21백만 톤
    - 터키: 약 13백만 톤
    - 미국: 약 10백만 톤
    - 이집트: 약 6백만 톤
    - 이탈리아: 약 5백만 톤
    """
    np.random.seed(42)
    years = list(range(2000, 2024))
    
    # 주요 토마토 생산국 및 기준 생산량 (백만 톤)
    countries_base = {
        'China': {'production': 60, 'growth': 0.025},      # 연 2.5% 성장
        'India': {'production': 18, 'growth': 0.03},       # 연 3% 성장
        'Turkey': {'production': 11, 'growth': 0.02},
        'United States of America': {'production': 12, 'growth': 0.005},
        'Egypt': {'production': 7, 'growth': 0.025},
        'Italy': {'production': 6, 'growth': 0.01},
        'Iran (Islamic Republic of)': {'production': 6, 'growth': 0.02},
        'Spain': {'production': 4, 'growth': 0.015},
        'Mexico': {'production': 4, 'growth': 0.02},
        'Brazil': {'production': 4, 'growth': 0.02}
    }
    
    records = []
    
    for country, info in countries_base.items():
        base_prod = info['production']
        growth = info['growth']
        
        for year in years:
            # 연도별 트렌드 + 기후 변동성
            trend = (year - 2000) * growth
            climate_shock = np.random.normal(0, 0.05)
            
            # 특정 기후 이벤트 반영
            if year == 2012 and country == 'United States of America':
                climate_shock = -0.12  # 2012 가뭄
            if year == 2019 and country == 'Italy':
                climate_shock = -0.08  # 유럽 폭염
            if year == 2022 and country in ['Spain', 'Italy']:
                climate_shock = -0.10  # 2022 유럽 가뭄
            
            production = base_prod * (1 + trend + climate_shock) * 1_000_000
            area = production / (np.random.uniform(40, 60) * 10000)  # 톤/헥타르
            yield_val = (production / area) * 10 if area > 0 else 0  # hg/ha
            
            records.extend([
                {'Area': country, 'Item': 'Tomatoes', 'Element': 'Production',
                 'Year': year, 'Unit': 'tonnes', 'Value': int(production)},
                {'Area': country, 'Item': 'Tomatoes', 'Element': 'Area harvested',
                 'Year': year, 'Unit': 'ha', 'Value': int(area)},
                {'Area': country, 'Item': 'Tomatoes', 'Element': 'Yield',
                 'Year': year, 'Unit': 'hg/ha', 'Value': int(yield_val)}
            ])
    
    return pd.DataFrame(records)


def create_eggs_production_sample():
    """
    계란 생산량 샘플 데이터 생성
    
    출처: FAO. 2024. FAOSTAT: Livestock Primary.
          https://www.fao.org/faostat/en/#data/QL
    
    주요 생산국 (2022년 기준):
    - 중국: 약 34백만 톤 (세계 1위, 35%)
    - 미국: 약 6.7백만 톤
    - 인도: 약 6.3백만 톤
    - 멕시코: 약 3백만 톤
    - 브라질: 약 2.8백만 톤
    """
    np.random.seed(43)
    years = list(range(2000, 2024))
    
    countries_base = {
        'China': {'production': 28, 'growth': 0.02},
        'United States of America': {'production': 5.5, 'growth': 0.015},
        'India': {'production': 3.5, 'growth': 0.04},
        'Mexico': {'production': 2.2, 'growth': 0.02},
        'Brazil': {'production': 2.0, 'growth': 0.025},
        'Japan': {'production': 2.5, 'growth': 0.005},
        'Russian Federation': {'production': 2.3, 'growth': 0.015},
        'Indonesia': {'production': 1.8, 'growth': 0.03},
        'Turkey': {'production': 1.2, 'growth': 0.025},
        'France': {'production': 0.9, 'growth': 0.01}
    }
    
    records = []
    
    for country, info in countries_base.items():
        base_prod = info['production']
        growth = info['growth']
        
        for year in years:
            trend = (year - 2000) * growth
            
            # 사료 가격/질병 영향 (간접적 기후 영향)
            shock = np.random.normal(0, 0.04)
            
            # AI(조류인플루엔자) 발생 연도 반영
            if year == 2015 and country == 'United States of America':
                shock = -0.08  # 2015 AI 발생
            if year == 2022 and country in ['United States of America', 'France']:
                shock = -0.06  # 2022 AI 발생
            
            # 사료 가격 급등 (옥수수/콩 흉작)
            if year == 2012 and country in ['United States of America', 'Mexico']:
                shock = -0.05  # 사료 가격 상승 영향
            
            production = base_prod * (1 + trend + shock) * 1_000_000
            
            records.append({
                'Area': country,
                'Item': 'Eggs, hen, in shell',
                'Element': 'Production',
                'Year': year,
                'Unit': 'tonnes',
                'Value': int(production)
            })
    
    return pd.DataFrame(records)

def create_temperature_sample():
    """
    기온 변화 데이터 샘플 생성
    
    출처: FAO. 2024. FAOSTAT: Temperature change on land.
          https://www.fao.org/faostat/en/#data/ET
    원본: NASA Goddard Institute for Space Studies (GISTEMP)
    """
    np.random.seed(45)
    years = list(range(1990, 2024))
    
    # 소스 원료 주요 생산국
    countries = ['World', 'China', 'India', 'United States of America', 
                 'Mexico', 'Turkey', 'Italy', 'Spain', 'Thailand', 'Korea, Republic of']
    
    regional_factors = {
        'World': 1.0,
        'China': 1.1,
        'India': 0.95,
        'United States of America': 1.2,
        'Mexico': 1.1,
        'Turkey': 1.15,
        'Italy': 1.2,
        'Spain': 1.25,
        'Thailand': 0.85,
        'Korea, Republic of': 1.1
    }
    
    records = []
    for country in countries:
        factor = regional_factors.get(country, 1.0)
        for year in years:
            base_anomaly = (year - 1990) * 0.025
            variation = np.random.normal(0, 0.12)
            
            temp_change = base_anomaly * factor + variation
            
            if year >= 2015:
                temp_change += 0.2
            if year >= 2020:
                temp_change += 0.15
            
            records.append({
                'Area': country,
                'Months': 'Meteorological year',
                'Element': 'Temperature change',
                'Year': year,
                'Unit': '°C',
                'Value': round(temp_change, 3)
            })
    
    return pd.DataFrame(records)


def create_korea_sauce_import_sample():
    """
    한국의 소스 원료 수입 데이터 샘플 생성
    
    출처: FAO. 2024. FAOSTAT: Crops and livestock products (Trade).
          https://www.fao.org/faostat/en/#data/TCL
    """
    np.random.seed(46)
    years = list(range(2000, 2024))
    
    items_base = {
        'Tomatoes': {'qty': 5_000, 'value': 8_000, 'growth': 0.04},
        'Tomato paste': {'qty': 30_000, 'value': 40_000, 'growth': 0.03},
        'Chillies and peppers, dry': {'qty': 80_000, 'value': 150_000, 'growth': 0.025},
        'Eggs, hen, in shell': {'qty': 3_000, 'value': 6_000, 'growth': 0.05}
    }
    
    records = []
    for item, info in items_base.items():
        for year in years:
            trend = (year - 2000) * info['growth']
            variation = np.random.normal(0, 0.1)
            
            qty = info['qty'] * (1 + trend + variation)
            value = info['value'] * (1 + trend + variation * 1.2)
            
            records.extend([
                {'Reporter Countries': 'Korea, Republic of', 'Partner Countries': 'World',
                 'Item': item, 'Element': 'Import Quantity', 'Year': year,
                 'Unit': 'tonnes', 'Value': int(qty)},
                {'Reporter Countries': 'Korea, Republic of', 'Partner Countries': 'World',
                 'Item': item, 'Element': 'Import Value', 'Year': year,
                 'Unit': '1000 US$', 'Value': int(value)}
            ])
    
    return pd.DataFrame(records)
```

### 5.1 생산량 데이터 전처리

```python
def preprocess_production(df, item_name, element='Production'):
    """
    생산량 데이터를 Wide format으로 변환
    
    Parameters:
    -----------
    df : DataFrame - 원본 FAOSTAT 데이터
    item_name : str - 품목명
    element : str - 요소 (Production, Area harvested, Yield)
    
    Returns:
    --------
    DataFrame - 연도 × 국가 피벗 테이블
    """
    mask = (df['Item'] == item_name) & (df['Element'] == element)
    df_filtered = df[mask].copy()
    
    if len(df_filtered) == 0:
        print(f"⚠️ '{item_name}' - '{element}' 데이터 없음")
        return None
    
    df_pivot = df_filtered.pivot_table(
        index='Year',
        columns='Area',
        values='Value',
        aggfunc='sum'
    ).reset_index()
    
    # 전체 합계
    numeric_cols = [c for c in df_pivot.columns if c != 'Year']
    df_pivot['World_Total'] = df_pivot[numeric_cols].sum(axis=1)
    
    return df_pivot


def preprocess_temperature(df):
    """기온 데이터 전처리"""
    df_annual = df[df['Months'] == 'Meteorological year'].copy()
    
    df_pivot = df_annual.pivot_table(
        index='Year',
        columns='Area',
        values='Value',
        aggfunc='mean'
    ).reset_index()
    
    return df_pivot


def merge_with_temperature(df_prod, df_temp, item_name):
    """생산량과 기온 데이터 병합"""
    if 'World' in df_temp.columns:
        df_merged = df_prod.merge(
            df_temp[['Year', 'World']].rename(columns={'World': 'Temp_Anomaly'}),
            on='Year', how='left'
        )
    else:
        df_merged = df_prod.copy()
        df_merged['Temp_Anomaly'] = np.nan
    
    df_merged['Item'] = item_name
    return df_merged
```

```python
def create_sauce_integrated_dataset():

        'Tomatoes (Ketchup)': 'tomatoes.csv',
        'Eggs (Mayonnaise)': 'eggs.csv',
        'Chillies_dry (Sriracha)': 'chillies_dry.csv',
        'Chillies_green': 'chillies_green.csv'
    }
    
    for item_name, filename in items_files.items():
        filepath = PROCESSED_DIR / filename
        if filepath.exists():
            df = pd.read_csv(filepath)
            
            for _, row in df.iterrows():
                integrated_data.append({
                    'Year': row['Year'],
                    'Item': item_name,
                    'World_Production_Tonnes': row.get('World_Total', np.nan),
                    'Temp_Anomaly_C': row.get('Temp_Anomaly', np.nan)
                })
    
    df_integrated = pd.DataFrame(integrated_data)
    df_integrated['YoY_Change_Pct'] = df_integrated.groupby('Item')['World_Production_Tonnes'].pct_change() * 100
    
    return df_integrated
```
